#include <iostream>
#include <ctime>

using namespace std;

int main() {
    const int ROWS = 5;
    const int COLS = 5;

    int arr[ROWS][COLS];

    srand(time(0));

    int* p = &arr[0][0];

    for (int i = 0; i < ROWS * COLS; i++) {
        *(p + i) = rand() % 100;
    }

    cout << "Array:\n";

    for (int i = 0; i < ROWS; i++) {
        for (int j = 0; j < COLS; j++) {
            cout << *(p + i * COLS + j) << "\t";
        }
        cout << endl;
    }

    cout << "\nMain diagonal:\n";
    for (int i = 0; i < ROWS; i++) {
        cout << *(p + i * COLS + i) << " ";
    }

    cout << "\n\nSecondary diagonal:\n";
    for (int i = 0; i < ROWS; i++) {
        cout << *(p + i * COLS + (COLS - 1 - i)) << " ";
    }

    int evenCount = 0;
    int oddCount = 0;

    for (int i = 0; i < ROWS * COLS; i++) {
        if (*(p + i) % 2 == 0)
            evenCount++;
        else
            oddCount++;
    }

    cout << "\n\nEven elements: " << evenCount;
    cout << "\nOdd elements: " << oddCount << endl;

    return 0;
}
